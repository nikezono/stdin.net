      $('#Refresh').click ->
    animateFlag = true
    animate = ->
      $("#Refresh").animate
        zIndex: 1
      ,
        duration: 500
        step: (now) ->
          return $(this).css transform: "rotate(" + (now * 360) + "deg)"
        complete: ->
          $("#Refresh").css('zIndex', 0)
          animate() if animateFlag
    animate()
    refresh ->
      animateFlag = false
